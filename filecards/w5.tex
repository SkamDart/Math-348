\begin{question}{1}
What is a solution to $42=17x+5y$ ?
\end{question}

\begin{proof}
$42=17x+5y$\\
$5y=42-17x$\\
$y=\frac{42-17x}{5}$\\
\text{let }$x=0$\\
$y=42$\\
\end{proof}
